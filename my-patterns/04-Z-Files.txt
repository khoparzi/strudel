  .scale('E minor')
  .transpose("<0 2 5 0>/4").transpose(-12)
  .off(1/8, x=>x.add("<7 7 7 7 7 <9 5 12>>"))

  .attack(tri.range(0.01, 0.1).slow(5.3))
  .cutoff(tri.range(600, 5000).slow(9)).resonance(0.25)
  .pianoroll({minMidi: 35})
